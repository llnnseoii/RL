1655function [Money, PaymentType] = AddCosts(Detected, CostStage, PaymentType, Money, time, z, mode)
1656% in this case a cancer has been diagnosed and
1657% we need to take care of the costs
1658
1659SubCost       = zeros(25, 404);  % for temporary calculations
1660SubCostFut    = zeros(25, 404);
1661SubCostAll    = zeros(1, 404);   %#ok<NASGU> % for temporary calculations
1662SubCostAllFut = zeros(1, 404);   %#ok<NASGU>
1663
1664Ende      = time;
1665l = length(find(Detected.Cancer(z, :)));
1666% we add the cost for the rest of the
1667% treatment time
1668for x1 = 1:l
1669    Start      = Detected.CancerYear(z, x1);
1670    Difference = Ende - Start;
1671    
1672    % costs differ dependent if patient died of tumor or of other causes
1673%     if isequal(mode, 'oc') % died of other causes
1674%         FinalCosts    = CostStage.Final_oc(Detected.Cancer(z, x1)-6); 
1675%         FutFinalCosts = CostStage.FutFinal_oc(Detected.Cancer(z, x1)-6); 
1676%     else
1677%         FinalCosts    = CostStage.Final(Detected.Cancer(z, x1)-6); % patient died of tumor
1678%         FutFinalCosts = CostStage.FutFinal(Detected.Cancer(z, x1)-6); % patient died of tumor
1679%     end
1680       
1681    % all other tumors
1682    %m for the costs, the following changes are made:
1683    %m after 5 years, the treatment is discontinued
1684    %m during the 5 years of treatment if the death is due to other
1685    %reasons, these costs are not included
1686    %m the initial phase is for 3 months, and the remaining time until the
1687    %terminal or the 5th year are the continuing costs.
1688    
1689    if  Difference <= 1/4
1690        % the costs for the first year of treatment apply, independent
1691        % whether the patient survived a full year or not
1692        SubCost(x1,  Start*4+1)             = CostStage.Initial(Detected.Cancer(z, x1)-6); % START
1693        SubCostFut(x1,  Start*4+1)          = CostStage.FutInitial(Detected.Cancer(z, x1)-6);
1694        PaymentType.Cancer_ini(Detected.Cancer(z, x1)-6, floor(Start)) = PaymentType.Cancer_ini(Detected.Cancer(z, x1)-6, floor(Start)) + 1;
1695        PaymentType.QCancer_ini(Detected.Cancer(z, x1)-6, floor(Start),1) = PaymentType.QCancer_ini(Detected.Cancer(z, x1)-6, floor(Start),1) + 1;
1696        
1697    elseif and((Difference > 1/4) , (Difference <= 1.25))
1698        SubCost(x1,  Start*4+1 )    = CostStage.Initial(Detected.Cancer(z, x1)-6);
1699        SubCostFut(x1,  Start*4+1 ) = CostStage.FutInitial(Detected.Cancer(z, x1)-6);
1700        PaymentType.Cancer_ini(Detected.Cancer(z, x1)-6, floor(Start)) = PaymentType.Cancer_ini(Detected.Cancer(z, x1)-6, floor(Start)) + 1;
1701        PaymentType.QCancer_ini(Detected.Cancer(z, x1)-6, floor(Start),1)= PaymentType.QCancer_ini(Detected.Cancer(z, x1)-6, floor(Start),1) + 1;
1702        
1703        if ~isequal(mode, 'oc')
1704            SubCost(x1, (Start*4+1)+1 : Ende*4)        = 1/4* CostStage.Final(Detected.Cancer(z, x1)-6); % START
1705            SubCostFut(x1, (Start*4+1)+1 : Ende*4)     = 1/4* CostStage.FutFinal(Detected.Cancer(z, x1)-6); % START
1706            PaymentType.Cancer_fin(Detected.Cancer(z, x1)-6, floor(Ende-1))  = PaymentType.Cancer_fin(Detected.Cancer(z, x1)-6, floor(Ende-1)) + (4*Ende-(4*Start+1))/4.0; %m2 ini is for q1, and the cont is for rest. but to keep it all integers, I use y
1707            for Qcount=1:(4*Ende-(4*Start+1))
1708                PaymentType.QCancer_fin(Detected.Cancer(z, x1)-6, floor(Ende-1),Qcount)  = PaymentType.QCancer_fin(Detected.Cancer(z, x1)-6, floor(Ende-1),Qcount) + 1; %m2 this finQ is counting in quarters
1709            end
1710        elseif isequal(mode, 'oc')
1711            SubCost(x1, (Start*4+1)+1 : Ende*4)        = 1/4* CostStage.Cont(Detected.Cancer(z, x1)-6); % START
1712            SubCostFut(x1, (Start*4+1)+1 : Ende*4)     = 1/4* CostStage.FutCont(Detected.Cancer(z, x1)-6); % START
1713            PaymentType.Cancer_con(Detected.Cancer(z, x1)-6, floor(Ende-1))  = PaymentType.Cancer_con(Detected.Cancer(z, x1)-6, floor(Ende-1)) + (4*Ende-(4*Start+1))/4.0; %m2 ini is for q1, and the cont is for rest. but to keep it all integers, I use y
1714            for Qcount=1:(4*Ende-(4*Start+1))
1715                PaymentType.QCancer_con(Detected.Cancer(z, x1)-6, floor(Ende-1),Qcount)  = PaymentType.QCancer_con(Detected.Cancer(z, x1)-6, floor(Ende-1),Qcount) + 1; %m2 this finQ is counting in quarters
1716            end
1717        end
1718    elseif and((Difference > 1.250) ,(Difference <= 5.0)) %m new costs treatment o
1719        SubCost(x1, (Start*4)+1 : (Start*4) + 1)    = CostStage.Initial(Detected.Cancer(z, x1)-6);
1720        SubCostFut(x1, Start*4+1 : (Start*4 + 1))   = CostStage.FutInitial(Detected.Cancer(z, x1)-6);
1721        SubCost(x1, (Start*4+1)+1 : (Ende-1)*4)     = 1/4*CostStage.Cont(Detected.Cancer(z, x1)-6); % CONT
1722        SubCostFut(x1, (Start*4+1)+1 : (Ende-1)*4)  = 1/4*CostStage.FutCont(Detected.Cancer(z, x1)-6); % CONT
1723        PaymentType.Cancer_ini(Detected.Cancer(z, x1)-6, floor(Start)) = PaymentType.Cancer_ini(Detected.Cancer(z, x1)-6,floor(Start)) + 1;
1724        PaymentType.QCancer_ini(Detected.Cancer(z, x1)-6, floor(Start),1)= PaymentType.QCancer_ini(Detected.Cancer(z, x1)-6, floor(Start),1) + 1;
1725        
1726        yyears = floor(Difference-1.25);
1727        qquarters = (Difference-1.25) - floor(Difference-1.25);
1728        
1729        for con_y=1:yyears
1730            PaymentType.Cancer_con(Detected.Cancer(z, x1)-6, floor(Start)+con_y) = PaymentType.Cancer_con(Detected.Cancer(z, x1)-6,floor(Start)+con_y) + 1;%((4*Ende-4*1)-(4*Start+1))/4;
1731            for Qcount=(con_y*4-3):(con_y*4)
1732                PaymentType.QCancer_con(Detected.Cancer(z, x1)-6, floor(Start)+con_y,Qcount) = PaymentType.QCancer_con(Detected.Cancer(z, x1)-6,floor(Start)+con_y,Qcount) + 1;%((4*Ende-4*1)-(4*Start+1))/4;
1733            end
1734        end
1735        
1736        PaymentType.Cancer_con(Detected.Cancer(z, x1)-6, floor(Start)+con_y+1)= PaymentType.Cancer_con(Detected.Cancer(z, x1)-6, floor(Start)+con_y+1) + qquarters;
1737        for Qcount=1:(4*qquarters)
1738            PaymentType.QCancer_con(Detected.Cancer(z, x1)-6, floor(Start)+con_y+1,con_y*4+Qcount) = PaymentType.QCancer_con(Detected.Cancer(z, x1)-6,floor(Start)+con_y+1,con_y*4+Qcount) + 1;%((4*Ende-4*1)-(4*Start+1))/4;
1739        end
1740        
1741        %m PaymentType.Cancer_con(Detected.Cancer(z, x1)-6, floor(Start)) = PaymentType.Cancer_con(Detected.Cancer(z, x1)-6,floor(Start)) + ((4*Ende-4*1)-(4*Start+1))/4;
1742        
1743        if ~isequal(mode, 'oc')
1744            SubCost(x1, (Ende-1)*4+1 : Ende*4)         = 1/4* CostStage.Final(Detected.Cancer(z, x1)-6);
1745            SubCostFut(x1, (Ende-1)*4+1 : Ende*4)      = 1/4* CostStage.FutFinal(Detected.Cancer(z, x1)-6);
1746            PaymentType.Cancer_fin(Detected.Cancer(z, x1)-6, floor(Ende)-1) = PaymentType.Cancer_fin(Detected.Cancer(z, x1)-6,floor(Ende)-1) + 1;
1747            for Qcount=1:4 %m here all 4 quarters of treatment are followed, unlike when difference<1.25 where only few quarters of final year are implemented
1748                PaymentType.QCancer_fin(Detected.Cancer(z, x1)-6, floor(Ende-1),Qcount)  = PaymentType.QCancer_fin(Detected.Cancer(z, x1)-6, floor(Ende-1),Qcount) + 1; %m2 this finQ is counting in quarters
1749            end
1750        elseif isequal(mode, 'oc')
1751            SubCost(x1, (Ende-1)*4+1 : Ende*4)         = 1/4* CostStage.Cont(Detected.Cancer(z, x1)-6);
1752            SubCostFut(x1, (Ende-1)*4+1 : Ende*4)      = 1/4* CostStage.FutCont(Detected.Cancer(z, x1)-6);
1753            PaymentType.Cancer_con(Detected.Cancer(z, x1)-6, floor(Ende)-1) = PaymentType.Cancer_con(Detected.Cancer(z, x1)-6,floor(Ende)-1) + 1;
1754            for Qcount=1:4%m8 (4*Ende-(4*Start+1)) %m here all 4 quarters of treatment are followed, unlike when difference<1.25 where only few quarters of final year are implemented
1755                PaymentType.QCancer_con(Detected.Cancer(z, x1)-6, floor(Ende-1),con_y*4+qquarters*4+Qcount)  = PaymentType.QCancer_con(Detected.Cancer(z, x1)-6, floor(Ende-1),con_y*4+qquarters*4+Qcount) + 1; %m2 this finQ is counting in quarters %m8
1756            end
1757        end
1758    elseif Difference > 5
1759        SubCost(x1, Start*4+1 : (Start*4 + 1))     = CostStage.Initial(Detected.Cancer(z, x1)-6);
1760        SubCostFut(x1, Start*4+1 : (Start*4 + 1))  = CostStage.FutInitial(Detected.Cancer(z, x1)-6);
1761       %m8 SubCost(x1, (Start*4+1)+1 : (Start+5)*4)        = 1/4*CostStage.Cont(Detected.Cancer(z, x1)-6); % CONT
1762       %m8 SubCostFut(x1, (Start*4+1)+1 : (Start+5)*4)     = 1/4*CostStage.FutCont(Detected.Cancer(z, x1)-6); % CONT
1763        SubCost(x1, (Start*4+1)+1 : Ende*4)    = 1/4*CostStage.Cont(Detected.Cancer(z, x1)-6); % CONT
1764        SubCostFut(x1, (Start*4+1)+1 : Ende*4) = 1/4*CostStage.FutCont(Detected.Cancer(z, x1)-6); % CONT
1765        PaymentType.Cancer_ini(Detected.Cancer(z, x1)-6, floor(Start)) = PaymentType.Cancer_ini(Detected.Cancer(z, x1)-6, floor(Start)) + 1;
1766        PaymentType.QCancer_ini(Detected.Cancer(z, x1)-6, floor(Start),1)= PaymentType.QCancer_ini(Detected.Cancer(z, x1)-6, floor(Start),1) + 1;
1767        %m the following con_y is for all Difference. it must be just upto
1768        %year 5. So, yyears must be 5?
1769        %%m yyears = floor(Difference-1.25);
1770        %%m qquarters = (Difference-1.25) - floor(Difference-1.25);
1771        for con_y=1:4%%m yyears
1772            PaymentType.Cancer_con(Detected.Cancer(z, x1)-6, floor(Start)+con_y) = PaymentType.Cancer_con(Detected.Cancer(z, x1)-6,floor(Start)+con_y) + 1;%((4*Ende-4*1)-(4*Start+1))/4;
1773            for Qcount=(con_y*4-3):(con_y*4)
1774                PaymentType.QCancer_con(Detected.Cancer(z, x1)-6, floor(Start)+con_y,Qcount) = PaymentType.QCancer_con(Detected.Cancer(z, x1)-6,floor(Start)+con_y,Qcount) + 1;%((4*Ende-4*1)-(4*Start+1))/4;
1775            end
1776        end
1777        PaymentType.Cancer_con(Detected.Cancer(z, x1)-6, floor(Start)+con_y+1)= PaymentType.Cancer_con(Detected.Cancer(z, x1)-6, floor(Start)+con_y+1) + 0.75;
1778        for Qcount=1:3
1779            PaymentType.QCancer_con(Detected.Cancer(z, x1)-6, floor(Start)+con_y+1,Qcount) = PaymentType.QCancer_con(Detected.Cancer(z, x1)-6,floor(Start)+con_y+1,Qcount) + 1;%((4*Ende-4*1)-(4*Start+1))/4;
1780        end
1781        
1782        %m PaymentType.Cancer_con(Detected.Cancer(z, x1)-6, floor(Start))  = PaymentType.Cancer_con(Detected.Cancer(z, x1)-6, floor(Start)) + (4*Ende-(4*Start+1))/4; %m2 ini is for q1, and the cont is for rest. but to keep it all integers, I use y
1783    end
1784    
1785end
1786
1787SubCostAll    =  sum(SubCost); %m2 sum(SubCost,1); remove max and add sum %max(SubCost)/4; % since all costs are quarterly we need to adjust %m3 and remove max()
1788SubCostAllFut =  sum(SubCostFut); % since all costs are quarterly we need to adjust %m3
1789Counter = 1;
1790for x1 = 1:100
1791    for x2= 1:4
1792        Money.Treatment(x1)       = Money.Treatment(x1) + SubCostAll(Counter);
1793        Money.FutureTreatment(x1) = Money.FutureTreatment(x1) + SubCostAllFut(Counter);
1794        Counter = Counter + 1;
1795    end
1796end
